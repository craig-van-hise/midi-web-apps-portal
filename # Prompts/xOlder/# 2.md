3. **Inject `usePortalLink` Hook:** Implement this exact pattern in each app:
```javascript
import { useEffect, useState } from 'react';

export function usePortalLink(onMidiIn, onPanic) {
  const [showInfo, setShowInfo] = useState(false);
  const [showSettings, setShowSettings] = useState(false);

  useEffect(() => {
    const handleMessage = (event) => {
      const { type, payload } = event?.data || {};
      if (type === 'VV_MIDI_IN' && onMidiIn) onMidiIn(payload);
      if (type === 'VV_TOGGLE_INFO') setShowInfo(prev => !prev);
      if (type === 'VV_TOGGLE_SETTINGS') setShowSettings(prev => !prev);
      if (type === 'VV_PANIC' && onPanic) onPanic();
    };
    window.addEventListener('message', handleMessage);
    return () => window.removeEventListener('message', handleMessage);
  }, [onMidiIn, onPanic]);

  const sendMidiOut = (midiData) => {
    window.parent.postMessage({ type: 'VV_MIDI_OUT', payload: midiData }, '*');
  };

  return { showInfo, showSettings, sendMidiOut, setShowInfo, setShowSettings };
}

```


4. **Wire the Logic:** * Connect `showInfo` and `showSettings` to trigger the app's existing Modals.

* Replace all local Tone.js audio generation with `sendMidiOut(data)`.
