## 1. Root Cause Analysis
The application is throwing `TypeError: Failed to construct 'AudioWorkletNode': parameter 1 is not of type 'BaseAudioContext'`. This occurs because Tone.js abstracts the audio context, and `Tone.context.rawContext` is failing the browser's strict prototype check. We must explicitly instantiate a raw `window.AudioContext` first, construct the Worklet with it, and then pass it to Tone.js to wrap.

    if (this.isInitialized || this.initPromise) return this.initPromise;

      let nativeCtx;
      if (!Tone.context || Tone.context.state !== 'running') {
        nativeCtx = new window.AudioContext({ latencyHint: 'interactive' });

        const optimizedContext = new Tone.Context(nativeCtx);
        optimizedContext.lookAhead = 0;
        optimizedContext.updateInterval = 0.01;
        Tone.setContext(optimizedContext);
      } else {
        nativeCtx = Tone.context.rawContext || Tone.getContext().rawContext;
      }

      try {
        await Tone.start();
      } catch (e) {
        console.warn("Tone.start() waiting for interaction...");
      }

      this.panVol = new Tone.PanVol(0, 0); 
      this.splitter = new Tone.Split(2);
      this.meterL = new Tone.Meter();
      this.meterR = new Tone.Meter();
      this.reverb = new Tone.Reverb({ decay: 2.5, preDelay: 0.01, wet: 0 });
      await this.reverb.generate(); 
      this.internalTrim = new Tone.Gain(1).connect(this.panVol);
      this.panVol.connect(this.splitter);
      this.splitter.connect(this.meterL, 0, 0);
      this.splitter.connect(this.meterR, 1, 0);
      this.panVol.chain(this.reverb, Tone.Destination);

      // 2. AudioWorklet Bridge using the GUARANTEED Native Context
      try {
        const basePath = import.meta.env?.BASE_URL || '/';
        const workletUrl = `${basePath}RomplerWorklet.js`.replace(/\/\//g, '/');
        
        await nativeCtx.audioWorklet.addModule(workletUrl);
        
        // nativeCtx is guaranteed to inherit from BaseAudioContext
        this.nativeWorklet = new AudioWorkletNode(nativeCtx, 'rompler-processor');
        
        // Because Tone.js overrides native connect, we can connect the native node to Tone nodes natively:
        this.nativeWorklet.connect(this.internalTrim.input || this.internalTrim);

        if (this.sabBuffer) {
          this.nativeWorklet.port.postMessage({ type: 'SAB', buffer: this.sabBuffer });
        }
        console.log('[AudioEngine] AudioWorklet bridge established successfully.');
      } catch (err) {
        console.error('[AudioEngine] Failed to initialize AudioWorklet:', err);
      }

      this.isInitialized = true;
    })();
    return this.initPromise;
  }
