
# Product Requirements Prompt (PRP) — ✅ Completed

## 1. Project Context & Objectives
The spelled notes strip above the `Keyboard` component suffers from text collision. When users input dense chords containing adjacent notes or chromatic clusters (e.g., D4 and Eb4), the text labels overlap horizontally. The rudimentary zipper logic only tracks a single `lastX` position, failing to account for string width or multi-note clusters. The objective is to replace this with a "Multi-Row Boundary Tracker" that estimates string widths, maintains horizontal boundaries across 3 distinct vertical rows, and safely slots each label to prevent overlap.

## 2. Technical Decisions & Dependencies
* **Target File:** `src/plugins/chord-notator/components/Keyboard.tsx`
* **Logic Specification:**
  * Define 3 vertical rows: Bottom (`75%`), Top (`25%`), Middle (`50%`).
  * Maintain an array `rowEdges = [-100, -100, -100]` representing the occupied right-boundary in pixels for each row.
  * For each note, calculate `approxWidth = (string.length * 6.5) + PADDING`.
  * Check the left edge of the new note against the `rowEdges` array to find the first row without a collision. If all collide (extreme clusters), fallback to the row with the minimal overflow.
  * Apply Tailwind utility classes `bg-white/80 dark:bg-black/80 backdrop-blur-[2px]` to the injected DOM nodes to ensure legibility during extreme density.

/**
 * Updates the spelled notes strip via direct DOM access
 * Aligns labels horizontally and vertically with the keys.
 * Implements a Multi-Row Boundary Tracker to prevent text overlap.
 */
export const updateSpelledNotesStrip = (spellings: { note: number; spelling: string }[], selectedNotes?: number[]) => {
    const el = document.getElementById('spelled-notes-strip');
    if (!el) return;
    
    // Clear existing
    el.innerHTML = '';

    // Dynamically adjust container border/background to reflect active selection
    const hasSelection = spellings.some(data => selectedNotes?.includes(data.note));
    if (hasSelection) {
        el.style.backgroundColor = 'rgba(170, 59, 255, 0.03)';
        el.style.borderColor = 'rgba(170, 59, 255, 0.1)';
    } else {
        el.style.backgroundColor = 'rgba(59, 130, 246, 0.03)';
        el.style.borderColor = 'rgba(59, 130, 246, 0.1)';
    }

    // 3-Row Zipper Tracking: [Bottom, Top, Middle]
    const rowEdges = [-100, -100, -100];
    const rowPositions = ['75%', '25%', '50%'];
    const PADDING = 6;

    spellings.forEach(data => {
        const x = getNoteX(data.note);
        
        // Approximate width based on character count (avg 6.5px per character for 10px Jost font)
        const approxWidth = data.spelling.length * 6.5 + PADDING;
        const leftEdge = x - (approxWidth / 2);
        const rightEdge = x + (approxWidth / 2);

        let assignedRow = -1;

        // 1. Try to find a row with no collision, prioritizing Bottom -> Top -> Middle
        for (let i = 0; i < rowEdges.length; i++) {
            if (leftEdge >= rowEdges[i]) {
                assignedRow = i;
                break;
            }
        }

        // 2. If all rows collide (extreme clusters), pick the row with the least overlap
        if (assignedRow === -1) {
            let minOverflow = Infinity;
            let bestRow = 0;
            for (let i = 0; i < rowEdges.length; i++) {
                const overflow = rowEdges[i] - leftEdge;
                if (overflow < minOverflow) {
                    minOverflow = overflow;
                    bestRow = i;
                }
            }
            assignedRow = bestRow;
        }

        // 3. Update the boundary edge tracker
        rowEdges[assignedRow] = rightEdge;

        const label = document.createElement('div');
        label.textContent = data.spelling;
        
        // Structure styling
        label.style.position = 'absolute';
        label.style.left = `${x}px`;
        label.style.top = rowPositions[assignedRow];
        label.style.transform = 'translate(-50%, -50%)';
        label.style.whiteSpace = 'nowrap';
        label.style.padding = '1px 3px';
        label.style.borderRadius = '3px';
        
        // Tailwind classes for contextual backgrounds to maintain legibility
        label.className = 'bg-white/80 dark:bg-[#0a0a0a]/80 backdrop-blur-[2px] transition-colors';
        
        // State styling
        const isSelected = selectedNotes?.includes(data.note);
        label.style.color = isSelected ? '#aa3bff' : '#3b82f6';
        if (isSelected) {
             label.style.fontWeight = '900';
             label.style.zIndex = '10';
        } else {
             label.style.zIndex = '5';
        }
        
        el.appendChild(label);
    });
};
